CMAKE_MINIMUM_REQUIRED(VERSION 2.6)

SET ( DATAEXCHANGESOURCES
	GBoolParameter
	GDataExchange
	GDoubleParameter
	GLongParameter
	GNumericParameterT
	GParameterValuePair
)

# BUILD_SHARED_LIBS controls the behavior of ADD_LIBRARY if STATIC/SHARED omitted
IF ( GENEVA_STATIC )
	ADD_LIBRARY ( ${DATAEXCHANGE_LIBNAME} STATIC ${DATAEXCHANGESOURCES} )
ELSE ()
	ADD_LIBRARY ( ${DATAEXCHANGE_LIBNAME} SHARED ${DATAEXCHANGESOURCES} )
ENDIF ( GENEVA_STATIC )

SET_TARGET_PROPERTIES( ${DATAEXCHANGE_LIBNAME}
	PROPERTIES VERSION ${VERSION_MAJOR}.${VERSION_MINOR}.${VERSION_PATCH}
)

INSTALL ( TARGETS ${DATAEXCHANGE_LIBNAME} DESTINATION lib
	PERMISSIONS
	OWNER_READ OWNER_WRITE
	GROUP_READ
	WORLD_READ
)
