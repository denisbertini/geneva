CMAKE_MINIMUM_REQUIRED(VERSION 2.8)

SET ( COMMONSOURCES
	GCanvas
	GCommonEnums
	GDefaultValueT
	GExceptions
	GFormulaParserT
	GHelperFunctions
	GLockVarT
	GLogger
	GMathHelperFunctions
	GParserBuilder
	GPlotDesigner
	GPODExpectationChecksT
	GThreadGroup
	GThreadPool
	GThreadWrapper
)

# BUILD_SHARED_LIBS controls the behavior of ADD_LIBRARY if STATIC/SHARED omitted
IF( GENEVA_STATIC )
	ADD_LIBRARY ( ${COMMON_LIBNAME} STATIC ${COMMONSOURCES} )
ELSE()
	ADD_LIBRARY ( ${COMMON_LIBNAME} SHARED ${COMMONSOURCES} )
ENDIF( GENEVA_STATIC )

SET_TARGET_PROPERTIES( ${COMMON_LIBNAME}
	PROPERTIES VERSION ${VERSION_MAJOR}.${VERSION_MINOR}.${VERSION_PATCH}
)

INSTALL ( TARGETS ${COMMON_LIBNAME} DESTINATION lib
	PERMISSIONS
	OWNER_READ OWNER_WRITE
	GROUP_READ
	WORLD_READ
)
