CMAKE_MINIMUM_REQUIRED(VERSION 2.6)

SET ( RANDOMSOURCES
	GRandom
	GRandomFactory
)

# BUILD_SHARED_LIBS controls the behavior of ADD_LIBRARY if STATIC/SHARED omitted
IF ( GENEVA_STATIC )
	ADD_LIBRARY ( ${RANDOMLIBNAME} STATIC ${RANDOMSOURCES} )
ELSE ()
	ADD_LIBRARY ( ${RANDOMLIBNAME} SHARED ${RANDOMSOURCES} )
ENDIF ( GENEVA_STATIC )

SET_TARGET_PROPERTIES( ${RANDOMLIBNAME}
	PROPERTIES VERSION ${VERSION_MAJOR}.${VERSION_MINOR}.${VERSION_PATCH}
)

INSTALL ( TARGETS ${RANDOMLIBNAME} DESTINATION lib
	PERMISSIONS
	OWNER_READ OWNER_WRITE
	GROUP_READ
	WORLD_READ
)
