CMAKE_MINIMUM_REQUIRED(VERSION 2.6)

SET ( COMMUNICATIONSOURCES
	GAsioHelperFunctions
)

# BUILD_SHARED_LIBS controls the behavior of ADD_LIBRARY if STATIC/SHARED omitted
IF ( GENEVA_STATIC )
	ADD_LIBRARY ( ${COMMUNICATIONLIBNAME} STATIC ${COMMUNICATIONSOURCES} )
ELSE ()
	ADD_LIBRARY ( ${COMMUNICATIONLIBNAME} SHARED ${COMMUNICATIONSOURCES} )
ENDIF ( GENEVA_STATIC )

INSTALL ( TARGETS ${COMMUNICATIONLIBNAME} DESTINATION lib
	PERMISSIONS
	OWNER_READ OWNER_WRITE
	GROUP_READ
	WORLD_READ
)
