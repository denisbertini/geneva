CMAKE_MINIMUM_REQUIRED(VERSION 2.6)

SET (EXECUTABLENAME GStartProjectUnitTests)

SET (${EXECUTABLENAME}_SRCS 
	GStartProjectUnitTests
)

ADD_EXECUTABLE(${EXECUTABLENAME}
	${${EXECUTABLENAME}_SRCS}
)

INCLUDE_DIRECTORIES (
	${PROJECT_SOURCE_DIR}
	${GENEVA_ROOT}/include
	${GENEVA_ROOT}/include/dataexchange
	${GENEVA_ROOT}/include/hap
	${GENEVA_ROOT}/include/courtier
	${GENEVA_ROOT}/include/geneva
	${GENEVA_ROOT}/tests/geneva/UnitTests/UtilityHeaders
#	${Boost_INCLUDE_DIRS}
)

LINK_DIRECTORIES (
	${GENEVA_ROOT}/lib
	${Boost_LIBRARY_DIRS}
)

TARGET_LINK_LIBRARIES (${EXECUTABLENAME}
	${GENEVA_LIBNAME}
	${GENEVA_INDIVIDUAL_LIBNAME}
	${COMMUNICATION_LIBNAME}
	${DATAEXCHANGE_LIBNAME}
	${HAP_LIBNAME}
	${COMMON_LIBNAME}
	${Boost_LIBRARIES}
)

INSTALL ( TARGETS ${EXECUTABLENAME} DESTINATION tests/Examples
	PERMISSIONS
	OWNER_READ OWNER_WRITE OWNER_EXECUTE
	GROUP_READ GROUP_EXECUTE
	WORLD_READ WORLD_EXECUTE
)
