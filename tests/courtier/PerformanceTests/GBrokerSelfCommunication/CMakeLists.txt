CMAKE_MINIMUM_REQUIRED(VERSION 2.8)

SET (EXECUTABLENAME GBrokerSelfCommunication)

SET (${EXECUTABLENAME}_SRCS
	GArgumentParser
	GBrokerSelfCommunication
	GBrokerSelfCommunicationEnums
	../../Misc/GRandomNumberContainer
	../../Misc/GSimpleContainer
)

SET ( CONFIGFILES
	GBrokerSelfCommunication.cfg
)

# Copy the config files to the build tree for easier testing
FOREACH ( file IN LISTS CONFIGFILES )
	CONFIGURE_FILE ( ${file} ${file} . COPYONLY )
ENDFOREACH ()

ADD_EXECUTABLE(${EXECUTABLENAME}
	${${EXECUTABLENAME}_SRCS}
)

INCLUDE_DIRECTORIES (
	${PROJECT_SOURCE_DIR}/tests/courtier/Misc
)

TARGET_LINK_LIBRARIES (${EXECUTABLENAME}
	${HAP_LIBNAME}
	${COURTIER_LIBNAME}
	${COMMON_LIBNAME}
	${Boost_LIBRARIES}
)

ADD_TEST(${EXECUTABLENAME} ${CMAKE_CURRENT_BINARY_DIR}/${EXECUTABLENAME})

INSTALL ( TARGETS ${EXECUTABLENAME} DESTINATION tests/courtier/PerformanceTests/${EXECUTABLENAME}
	PERMISSIONS
	OWNER_READ OWNER_WRITE OWNER_EXECUTE
	GROUP_READ GROUP_EXECUTE
	WORLD_READ WORLD_EXECUTE
)

INSTALL ( FILES ${CONFIGURATIONFILES} DESTINATION tests/courtier/PerformanceTests/${EXECUTABLENAME} )
