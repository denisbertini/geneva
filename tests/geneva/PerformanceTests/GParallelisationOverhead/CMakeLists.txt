CMAKE_MINIMUM_REQUIRED(VERSION 2.6)

SET (EXECUTABLENAME GParallelisationOverhead)

SET (${EXECUTABLENAME}_SRCS
	GParallelisationOverhead
	GDelayIndividual
)

SET (CONFIGURATIONFILES
    GParallelisationOverhead.cfg
    GDelayIndividual.cfg
)

ADD_EXECUTABLE(${EXECUTABLENAME}
	${${EXECUTABLENAME}_SRCS}
)

# Copy the config files to the build tree for easier testing
FOREACH ( file IN LISTS CONFIGURATIONFILES )
	CONFIGURE_FILE ( ${file} . COPYONLY )
ENDFOREACH ()

TARGET_LINK_LIBRARIES (${EXECUTABLENAME}
	${HAP_LIBNAME}
	${COURTIER_LIBNAME}
	${GENEVA_LIBNAME}
	${GENEVA_INDIVIDUAL_LIBNAME}
	${COMMON_LIBNAME}
	${Boost_LIBRARIES}
)

INSTALL ( TARGETS ${EXECUTABLENAME} DESTINATION tests/geneva/PerformanceTests/${EXECUTABLENAME}
	PERMISSIONS
	OWNER_READ OWNER_WRITE OWNER_EXECUTE
	GROUP_READ GROUP_EXECUTE
	WORLD_READ WORLD_EXECUTE
)

INSTALL ( FILES ${CONFIGURATIONFILES} DESTINATION tests/geneva/PerformanceTests/${EXECUTABLENAME}
	PERMISSIONS
	OWNER_READ OWNER_WRITE
	GROUP_READ
	WORLD_READ
)
