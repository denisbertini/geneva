CMAKE_MINIMUM_REQUIRED(VERSION 2.6)

SET (EXECUTABLENAME GNeuralNetworkUnitTests)

SET (${EXECUTABLENAME}_SRCS 
	GNeuralNetworkUnitTests
	../../GNeuralNetworkIndividual
)

ADD_EXECUTABLE(${EXECUTABLENAME}
	${${EXECUTABLENAME}_SRCS}
)

INCLUDE_DIRECTORIES (
        ${GENEVA_ROOT}/include
        ${GENEVA_ROOT}/tests/UnitTests/UtilityHeaders
        ${CMAKE_HOME_DIRECTORY} # The top of the source tree
        ${Boost_INCLUDE_DIRS}
)

TARGET_LINK_LIBRARIES (${EXECUTABLENAME}
	${GENEVA_LIBRARY}
	${Boost_LIBRARIES}
	pthread
)

LINK_DIRECTORIES (
        ${GENEVA_ROOT}/lib
        ${Boost_LIBRARY_DIRS}
)

INSTALL ( TARGETS ${EXECUTABLENAME} DESTINATION .
	PERMISSIONS
	OWNER_READ OWNER_WRITE OWNER_EXECUTE
	GROUP_READ GROUP_EXECUTE
	WORLD_READ WORLD_EXECUTE
)
