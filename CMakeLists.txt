PROJECT (geneva)
CMAKE_MINIMUM_REQUIRED(VERSION 2.6)

# Define our project version
SET( ${PROJECT_NAME}_MAJOR_VERSION 0 )
SET( ${PROJECT_NAME}_MINOR_VERSION 5 )
SET( ${PROJECT_NAME}_PATCH_LEVEL 0 )

# Enforce proper C++
ADD_DEFINITIONS ("-O3 -Wall -ansi -pthread")

# You can hardcode the paths here, if desired 
#SET (BOOST_ROOT "/opt/boost136")
#SET (BOOST_INCLUDEDIR /opt/boost136/include/boost-1_36/)
#SET (BOOST_LIBRARYDIR /opt/boost136/lib)

SET (Boost_USE_STATIC_LIBS OFF)
SET (Boost_USE_MULTITHREAD ON)

FIND_PACKAGE( Boost 1.36 REQUIRED COMPONENTS date_time system thread serialization)

INCLUDE_DIRECTORIES ( 
${PROJECT_SOURCE_DIR}/include 
${Boost_INCLUDE_DIRS}
)

LINK_DIRECTORIES (
	${PROJECT_SOURCE_DIR}/src
	${Boost_LIBRARY_DIRS}	
)

ADD_SUBDIRECTORY (src)
ADD_SUBDIRECTORY (include)
ADD_SUBDIRECTORY (examples)
ADD_SUBDIRECTORY (tests)

SET ( INFORMATIONFILES
    AUTHORS
    CHANGES
    COPYING
    INSTALL
    README
)

INSTALL ( FILES ${INFORMATIONFILES} DESTINATION .
	PERMISSIONS
	OWNER_READ OWNER_WRITE
	GROUP_READ
	WORLD_READ
)
